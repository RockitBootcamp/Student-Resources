### Get details of Invoice

```
// Invoice Controller
function detailAction($id) {
	$lineItems = DB::getLineItems();

	$total = 0;
	foreach($items as $item) {
		$total = $total + $item->subTotal;
	}

	// Hand off to view

	return View("invoiceDetail",
		["lineItems" => $lineItems,
		"total" =>$total,
		"items"=?$items 
		]
	);
}
```
